const SHEET_ID = "1vSsovVy4y7I5BFZJnmCFK7hpaeWkNHT8oX4ozZ10z5A"; function doGet() { return ContentService .createTextOutput("KAMU PRESET API Running") .setMimeType(ContentService.MimeType.TEXT); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); // SIGNUP if (data.action === "signup") { const sheet = ss.getSheetByName("Users"); sheet.appendRow([ data.username, data.password, data.whatsapp, new Date() ]); return ContentService .createTextOutput(JSON.stringify({ success: true, message: "Account Created" })) .setMimeType(ContentService.MimeType.JSON); } // LOGIN if (data.action === "login") { const sheet = ss.getSheetByName("Users"); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if ( rows[i][0] == data.username && rows[i][1] == data.password ) { return ContentService .createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ success: false, message: "Wrong Username Or Password" })) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({ success: false, message: "Invalid Action" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
